  -- Disable/enable constraints ONLY for our project tables
  PROCEDURE toggle_constraints(p_enable BOOLEAN) IS
  BEGIN
    FOR r IN (
      SELECT table_name, constraint_name 
      FROM user_constraints 
      WHERE constraint_type = 'R'
        AND table_name IN (
          'USER_REQUEST_QUOTAS', 'ADMIN_AUDIT_LOGS', 'ANALYSIS_DRAFTS', 
          'DRUG_EFFECTIVE_ENTRIES', 'DRUG_EXISTING_DETAILS', 'DRUG_ALTERNATIVE_NEGOTIATIONS', 
          'DRUG_ALTERNATIVES', 'AUDIT_LOGS', 'NOTIFICATIONS', 'REJECTION_REMARK_HISTORY', 
          'APPROVAL_REMARK_HISTORY', 'BLACKLISTED_COMPANIES', 'DRUG_REQUESTS', 'USERS', 'ADMIN_USERS'
        )
    ) LOOP
      IF p_enable THEN
        EXECUTE IMMEDIATE 'ALTER TABLE "' || r.table_name || '" ENABLE CONSTRAINT "' || r.constraint_name || '"';
      ELSE
        EXECUTE IMMEDIATE 'ALTER TABLE "' || r.table_name || '" DISABLE CONSTRAINT "' || r.constraint_name || '"';
      END IF;
    END LOOP;
  END;
BEGIN
  -- 1. Disable constraints only for our tables
